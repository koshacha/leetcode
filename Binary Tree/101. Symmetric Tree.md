# 101. Symmetric Tree

Given the `root` of a binary tree, check whether it is a mirror of itself (i.e., symmetric around its center).

## First Solution

- time: O(n)
- mem: O(n) ðŸ’©

```ts
/**
 * Definition for a binary tree node.
 * class TreeNode {
 *     val: number
 *     left: TreeNode | null
 *     right: TreeNode | null
 *     constructor(val?: number, left?: TreeNode | null, right?: TreeNode | null) {
 *         this.val = (val===undefined ? 0 : val)
 *         this.left = (left===undefined ? null : left)
 *         this.right = (right===undefined ? null : right)
 *     }
 * }
 */

function isSymmetric(root: TreeNode | null): boolean {
  if (!root || (!root.left && !root.right)) return true;

  const left = traversal(root.left);
  const right = traversal(root.right, true);

  if (left.length !== right.length) return false;

  for (let i in left) {
    if (left[i] !== right[i]) return false;
  }

  return true;
}

function traversal(root: TreeNode | null, mirrored = false): number[] {
  if (!root) return [null];
  const left = traversal(root.left, mirrored);
  const right = traversal(root.right, mirrored);

  return mirrored
    ? [...left, ...right, root.val]
    : [...right, ...left, root.val];
}
```
